        { id: 'v2', name: 'v0.9.0', createdAt: '2024-01-10 09:15:00', status: VERSION_STATUS.PUBLISHED },

    ]
};

// 版本切换函数
function handleVersionChange(versionId) {
    versionData.currentVersion = versionId;
    updateVersionButtons();
    showToast(`已切换到版本: ${versionId}`, 'success');
}

// 根据版本状态更新按钮
function updateVersionButtons() {
    const currentVersion = versionData.versions.find(v => v.id === versionData.currentVersion);
    if (!currentVersion) return;
    
    const status = currentVersion.status;
    const actionBar = document.querySelector('.action-bar');
    
    let buttons = '';
    if (status === VERSION_STATUS.DRAFT) {
        buttons = `
            <button class="btn" onclick="handleBack()">返回</button>
            <button class="btn" onclick="handleSaveDraft()">保存草稿</button>
            <button class="btn btn-primary" onclick="handlePublish()">发布</button>
        `;
    } else if (status === VERSION_STATUS.PUBLISHED) {
        buttons = `
            <button class="btn" onclick="handleBack()">返回</button>
            <button class="btn" onclick="handleNewDraft()">新增草稿版本</button>
            <button class="btn" onclick="handleInvalidate()">失效</button>
        `;
    } else if (status === VERSION_STATUS.INVALID) {
        buttons = `
            <button class="btn" onclick="handleBack()">返回</button>
            <button class="btn" onclick="handleNewDraft()">新增草稿版本</button>
            <button class="btn" style="color: var(--error-color);" onclick="handleDeleteVersion()">删除</button>
        `;
    }
    
    actionBar.innerHTML = buttons;
}

// 处理认证类型变化
function handleAuthTypeChange() {
    const checkboxes = document.querySelectorAll('#authTypeGroup input[type="checkbox"]');
    selectedAuthTypes = [];
    
    checkboxes.forEach(checkbox => {
        if (checkbox.checked) {
            selectedAuthTypes.push(checkbox.value);
        }
    });
    
    renderAuthParams();
}

// 渲染认证参数
function renderAuthParams() {
    const container = document.getElementById('authParamsContainer');
    
    if (selectedAuthTypes.length === 0) {
        container.innerHTML = '<p style="color: var(--text-tertiary); font-size: 13px;">请先选择认证方式</p>';
        return;
    }
    
    let html = '';
    
    selectedAuthTypes.forEach(authType => {
        const authSchema = AUTH_SCHEMA_MAP[authType] || [];
        const isSignature = authType === 'SIGNATURE';
        
        html += `
            <div class="auth-type-section" data-type="${authType}">
                <h4 style="font-size: 14px; margin-bottom: 12px; color: var(--text-primary);">
                    ${getAuthTypeName(authType)}
                </h4>
        `;
        
        authSchema.forEach((param, index) => {
            const isFixed = param.isFixed || isSignature;
            
            html += `
                <div class="form-group" style="margin-bottom: 12px;">
                    <div style="display: flex; gap: 12px; align-items: center; flex-wrap: wrap;">
                        <input type="text"
                               style="flex: 1; min-width: 120px; padding: 6px 10px; border: 1px solid var(--border-color); border-radius: 4px; font-size: 13px; ${isFixed ? 'background: var(--bg-light);' : ''}"
                               value="${param.paramName}"
                               ${isFixed ? 'disabled' : ''}
                               placeholder="参数名称">
                        <select style="padding: 6px 10px; border: 1px solid var(--border-color); border-radius: 4px; font-size: 13px; ${isFixed ? 'background: var(--bg-light);' : ''}">
                            <option value="header" ${param.carrier === 'header' ? 'selected' : ''}>header</option>
                            <option value="body" ${param.carrier === 'body' ? 'selected' : ''}>body</option>
                            <option value="query" ${param.carrier === 'query' ? 'selected' : ''}>query</option>
                        </select>
                        <input type="text"
                               style="flex: 2; min-width: 150px; padding: 6px 10px; border: 1px solid var(--border-color); border-radius: 4px; font-size: 13px;"
                               placeholder="参数值（${isSignature ? '固定值' : '可选'}）">
                        <input type="text"
                               style="flex: 2; min-width: 150px; padding: 6px 10px; border: 1px solid var(--border-color); border-radius: 4px; font-size: 13px;"
                               placeholder="参数描述">
                    </div>
                </div>
            `;
        });
        
        html += '</div>';
    });
    
    container.innerHTML = html;
}

// 获取认证类型名称
function getAuthTypeName(type) {
    const names = {
        SOA: 'SOA认证',
        APIG: 'APIG认证',
        COOKIE: 'Cookie认证',
        SIGNATURE: '数字签名认证'
    };
    return names[type] || type;
}

// 处理参数类型变更（增强版）
function handleParamTypeChange(selectElement) {
    const paramItem = selectElement.closest('.param-item');
    const newType = selectElement.value;
    const isComplex = newType === 'object' || newType === 'array';
    
    // 移除现有的子参数容器和添加按钮
    const existingChildren = paramItem.querySelector('.children-container');
    const existingAddBtn = paramItem.querySelector('.child-actions');
    if (existingChildren) existingChildren.remove();
    if (existingAddBtn) existingAddBtn.remove();
    
    // 如果是复杂类型，添加子参数容器和添加下拉菜单
    if (isComplex) {
        // 创建子参数容器
        const childrenContainer = document.createElement('div');
        childrenContainer.className = 'children-container';
        childrenContainer.style.cssText = 'margin-left: 24px; margin-top: 8px; padding-left: 12px; border-left: 2px solid var(--border-color);';
        paramItem.appendChild(childrenContainer);
        
        // 创建添加下拉菜单
        const addBtn = document.createElement('div');
        addBtn.className = 'child-actions';
        addBtn.style.cssText = 'margin-top: 8px; margin-left: 24px;';
        addBtn.innerHTML = `
            <div class="dropdown-menu" style="position: relative; display: inline-block;">
                <button type="button" class="add-child-btn" onclick="toggleChildDropdown(this)" 
                    style="padding: 6px 12px; border: 1px dashed var(--border-color); background: transparent; border-radius: 4px; cursor: pointer; font-size: 13px; color: var(--text-secondary);">
                    + 添加 <span style="margin-left: 4px;">▼</span>
                </button>

                        添加子参数
                    </button>
                    <button type="button" onclick="addChildParam(this, 'sibling')" style="display: block; width: 100%; padding: 8px 12px; border: none; background: transparent; text-align: left; cursor: pointer; font-size: 13px;">
                        添加同级参数
                    </button>
                </div>

// 切换下拉菜单
function toggleChildDropdown(btn) {
    const dropdown = btn.closest('.dropdown-menu').querySelector('.dropdown-content');
    dropdown.style.display = dropdown.style.display === 'block' ? 'none' : 'block';
}

// 添加子参数或同级参数
function addChildParam(btn, type) {
    // 关闭下拉菜单
    btn.closest('.dropdown-content').style.display = 'none';
    
    const paramItem = btn.closest('.param-item');
    const carrier = paramItem.dataset.carrier || 'body';
    
    if (type === 'child') {
        // 添加到children-container
        const childrenContainer = paramItem.querySelector('.children-container');
        if (childrenContainer) {
            const childHtml = createParamItemWithCarrier({}, carrier);
            childrenContainer.insertAdjacentHTML('beforeend', childHtml);
        }
    } else {
        // 添加到同级（parent的children-container或主容器）
        const parent = paramItem.parentElement;
        if (parent.classList.contains('children-container')) {
            parent.insertAdjacentHTML('beforeend', createParamItemWithCarrier({}, carrier));
        } else {
            parent.insertAdjacentHTML('beforeend', createParamItemWithCarrier({}, carrier));
        }
    }
}

// 切换更多菜单
function toggleMoreMenu(btn, flowId) {
    // 关闭其他菜单
    document.querySelectorAll('.more-menu').forEach(menu => {
        if (menu !== btn.nextElementSibling) {
            menu.style.display = 'none';
        }
    });
    
    const menu = btn.nextElementSibling;

function saveNodeConfig() {
    if (!selectedNode) return;

    selectedNode.data.label = document.getElementById('nodeLabel')?.value || '';
    selectedNode.data.description = document.getElementById('nodeDesc')?.value || '';

    // 根据类型保存特定配置
    if (selectedNode.type === 'trigger') {
        selectedNode.data.triggerType = document.getElementById('nodeTriggerType')?.value;
        
        // 保存SYSTOKEN凭证
        const systokenInputs = document.querySelectorAll('.systoken-item input');
        selectedNode.data.systokens = [];
        for (let i = 0; i < systokenInputs.length; i += 2) {
            if (systokenInputs[i]?.value && systokenInputs[i + 1]?.value) {
                selectedNode.data.systokens.push({
                    name: systokenInputs[i].value,
                    value: systokenInputs[i + 1].value
                });
            }
        }
        
        // 保存入参配置（用于调试）
        selectedNode.data.inputParams = [
            { name: 'param1', type: 'string' },
            { name: 'param2', type: 'number' }
        ];
        
    } else if (selectedNode.type === 'action') {
        selectedNode.data.connectorId = document.getElementById('nodeConnector')?.value;
        
    } else if (selectedNode.type === 'condition') {
        selectedNode.data.condition = document.getElementById('nodeCondition')?.value;
        
    } else if (selectedNode.type === 'data') {
        selectedNode.data.transform = document.getElementById('nodeDataTransform')?.value;
        selectedNode.data.outputField = document.getElementById('nodeOutputField')?.value;
        
    } else if (selectedNode.type === 'output') {
        selectedNode.data.outputFormat = document.getElementById('nodeOutputFormat')?.value;
    }

    closeDrawer();
    renderCanvas();
    showToast('配置已保存', 'success');
}
